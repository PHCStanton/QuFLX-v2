```python
def _calculate_cci(self, df: pd.DataFrame) -> pd.DataFrame:
    try:
        period = self.params['cci_period']  # 20
        typical_price = (df['high'] + df['low'] + df['close']) / 3
        sma_tp = typical_price.rolling(window=period).mean()
        
        # ❌ PROBLEM: Inefficient loop + NaN handling issue
        mean_dev = pd.Series(index=df.index, dtype=float)
        for i in range(period - 1, len(df)):
            deviations = []
            for j in range(i - period + 1, i + 1):
                deviations.append(abs(typical_price.iloc[j] - sma_tp.iloc[i]))
            mean_dev.iloc[i] = np.mean(deviations)
        
        cci = pd.Series(index=df.index, dtype=float)
        for i in range(len(df)):
            if mean_dev.iloc[i] != 0:
                cci.iloc[i] = (typical_price.iloc[i] - sma_tp.iloc[i]) / (0.015 * mean_dev.iloc[i])
            else:
                cci.iloc[i] = 0  # ← Division edge case handled, but NaN not
```

### __Issues Identified:__

1. __NaN Propagation__ - `mean_dev` starts as NaN for indices 0 to `period-2`, but the CCI loop iterates from 0, causing NaN output for early values
2. __Division by Zero Handling__ - Checks for `!= 0` but doesn't check for `pd.isna()`
3. __Performance__ - Double nested loop is O(n²), very slow for large datasets
4. __CORE_PRINCIPLES #9 Violation__ - No early validation of `typical_price` or `sma_tp` for NaN/inf values

```python
# RECOMMENDED FIX for _calculate_cci():
def _calculate_cci(self, df: pd.DataFrame) -> pd.DataFrame:
    try:
        period = self.params['cci_period']
        
        # 1. Calculate typical price
        typical_price = (df['high'] + df['low'] + df['close']) / 3
        
        # 2. SMA of typical price
        sma_tp = typical_price.rolling(window=period).mean()
        
        # 3. Mean deviation (vectorized, not looped!)
        mean_dev = typical_price.rolling(window=period).apply(
            lambda x: np.mean(np.abs(x - x.mean())), raw=True
        )
        
        # 4. CCI with safe division
        cci = (typical_price - sma_tp) / (0.015 * mean_dev.replace(0, np.nan))
        
        df['cci'] = cci
        
    except Exception as e:
        self.logger.error(f"Error calculating CCI: {str(e)}")
        df['cci'] = np.nan  # Explicit NaN, not None
    
    return df
```

__Why This Is Better:__

- ✅ Vectorized (O(n) vs O(n²))
- ✅ Handles NaN properly via pandas rolling
- ✅ Uses `replace(0, np.nan)` to avoid division issues
- ✅ Follows CORE_PRINCIPLES #1 (Functional Simplicity)
